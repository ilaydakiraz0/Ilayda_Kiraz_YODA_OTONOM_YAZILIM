SAYI TAHMİN

```python
import random

sayi = random.randint(1, 100) 
hak = 10
deneme = 0  

while hak > 0:  #hakkı sıfır olana kadar döngü devam edecek
    tahmin = int(input("1-100 arasinda tahmin yap: "))
    deneme += 1  
    hak -= 1     

    if tahmin == sayi:
        print(f"{deneme}. denemede doğru bildin. ")
        break
    elif tahmin < sayi:
        print("Daha yüksek bir tahmin yap. ")
    else:
        print("Daha düşük bir tahmin yap. ")

    print(f"{hak} hakkin kaldi\n")

if hak == 0:
    print(f"Hakkin bitti. Doğru sayi: {sayi}")
```

BANKA HESABI
```python
class Hesap:
    def __init__ (self, user, hesapno, bakiye=0):
        self.user= user
        self.hesapno= hesapno
        self.bakiye= bakiye

#hesaba para gelme işlemi
    def parayatir(self, tutar):
        if tutar>0: 
            self.bakiye += tutar 
            print(f"{tutar} TL hesabiniza geldi. Yeni bakiye {self.bakiye} TL")
        else:
            print("Geçersiz tutar.")

#hesaptan para çekme işlemi
    def paracekme(self, tutar):
        if tutar <=0:
            print("Geçersiz tutar")
        elif tutar> self.bakiye:
            print("Yetersiz bakiye")
        else:
            self.bakiye -= tutar
            print(f"{tutar} TL çekildi. Kalan bakiye: {self.bakiye}")
    
    def bakiyesorgu(self):
        print(f"{self.user} kullanicisinin bakiyesi: {self.bakiye}")

ad= input("Kullanıcı adı girin: ")
hesapno= input("Hesap numaranızı girin: ")
hesap= Hesap(ad, hesapno)

while True:
    print("1. Para yatır")
    print("2. Para çek")
    print("3. Bakiye sorgula")
    print("4. Çıkış")

    secim= input("1-4 ile işlem seçiniz.")

    if secim == "1":
        tutar= float(input("Yatırmak istediğiniz tutarı girin:"))
        hesap.parayatir(tutar)

    elif secim == "2":
        tutar= float(input("Hesabınızdan çekmek istediğiniz tutarı girin:"))
        hesap.paracekme(tutar)

    elif secim == "3":
        hesap.bakiyesorgu()

    elif secim == "4":
        print("Çıkış yapıldı.")
        break

    else:
        print("Geçerli sayıyı giriniz. (1-4)")
```
