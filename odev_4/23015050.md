## Tutorial 1
Open Cv'ye görüntü yüklemeyi, bu görüntülerin boyutunu ayarlamayı, döndürmeyi, yeni bir adla yeniden 
kaydetmeyi ve son olarak düzenlenmiş resmi ekranda göstermeyi öğrendim.

```python

import cv2

img = cv2.imread('C:/Users/ukugu/Desktop/yoda_opencv/images.png', -1 )
# cv2.IMREAD_COLOR: -1, arka plan transparanlığı yok eder ama olduğu gibi gösterir
# cv2.IMREAD_GRAYSCALE: 0, resmi siyah beyaz yapar
# cv2.IMREAD_UNCHANGED: 1, arka plan transpanlığı dahil olduğu gibi gösterir
img = cv2.resize(img, (400, 400))
img = cv2.resize(img, (0, 0), fx=0.5, fy=0.5)
img = cv2.rotate(img, cv2.ROTATE_90_CLOCKWISE)

cv2.waitKey(0) #görüntünün kaç milisaniye açık kalacağını belirler, 0 ise bir tuşa basana kadar açık kalır
cv2.destroyAllWindows() #açık olan görüntüleri kapatır

```

## Tutorial 4

Görüntüyü dört parçaya bölüp, bu parçaları döndürdüm.

```python

import numpy as np
import cv2

cap = cv2.VideoCapture(0)

while True: #kamera sürekli çalışsın diye sonsuz döngü
    ret, frame = cap.read()
    width = int(cap.get(3)) #fonksiyonlar float döndürüyor ama pikseller tam sayı olmak zorunda
    height = int(cap.get(4))

    image= np.zeros(frame.shape, np.uint8) #boş siyah görüntü oluşturma
    smaller_frame = cv2.resize(frame, (0, 0), fx=0.5, fy=0.5)
    image[:height//2, :width//2] = cv2.rotate(smaller_frame,cv2.ROTATE_180) #sol üst
    image[height//2:, :width//2] = smaller_frame #sol alt
    image[:height//2, width//2:] = cv2.rotate(smaller_frame,cv2.ROTATE_180) #sağ üst
    image[height//2:, width//2:] = smaller_frame #sağ alt
    
    cv2.imshow('frame', image)

    if cv2.waitKey(1) == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
```

## Tutorial 5

Kameradan aldığım görüntü üzerine çizgi ve dikdörtgen çizdim, metin ekledim. 

```python

import numpy as np
import cv2

cap = cv2.VideoCapture(0)

while True:
    ret, frame = cap.read()
    width = int(cap.get(3))
    height = int(cap.get(4))

    img = cv2.line(frame, 
     (0, 0), #çizginin başlangıç noktası (sol üst köşe)
     (width, height), #çizginin bitiş noktası (sağ alt köşe)
     (255, 0, 0 ), #renk (BGR formatında mavi)
     10) #çizgi kalınlığı
    
    img = cv2.rectangle(img, (100, 100), (200, 200), (0, 255, 0), 5) #dikdörtgen çizme
    font = cv2.FONT_HERSHEY_SIMPLEX
    img = cv2.putText(img, '<3', (200, height - 10), font, 4, (0, 0, 0), 5, cv2.LINE_AA)

    cv2.imshow('frame', img)

    if cv2.waitKey(1) == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
```

## Tutorial 6
Kameradan aldığım görüntü üzerinde renk tonlarını(bu projede mavi) tespit ettim ve HSV renk uzayında
gösterdim.

```python

import numpy as np
import cv2

cap = cv2.VideoCapture(0)

while True:
    ret, frame = cap.read()
    width = int(cap.get(3))
    height = int(cap.get(4))

    hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV) #BGR HSV'ye döner
    #H-saf renk, S-doygunluk, V-parlaklık. H 110-130 arasındaki mavi tonlar, 50-255 doygunluk ve parlaklık için)
    lower_blue = np.array([110, 50, 50]) 
    upper_blue = np.array([130, 255, 255])

    mask = cv2.inRange(hsv, lower_blue, upper_blue) #HSV görüntüsündeki her pikseli kontrol eder, 
    #lower_blue<=piksel<=upper_blue ise 1 yani beyaz, değilse 0 gösterir.
    
    result = cv2.bitwise_and(frame, frame, mask=mask) #orijinal görüntüde mavi olan yerleri göster

    cv2.imshow('frame', result) #mavi yerler mavi, kalanı siyah gösterir
    cv2.imshow('frame2', mask) #siyah-beyaz görüntüyü gösterir

    if cv2.waitKey(1) == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
```
## Tutorial 7

Satranç tahtası resmi kullanarak karelerin köşelerini tespit ettim ve her köşeden diğer köşelere
bağladım.
```python

import numpy as np
import cv2

img = cv2.imread('C:/Users/ukugu/Desktop/yoda_opencv/Chess_Board.png')
img = cv2.resize(img, (0,0), fx=0.75, fy=0.75)
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

corners = cv2.goodFeaturesToTrack(gray,
100, #bulunacak max köşe sayısı
0.01, #net köşeleri belirlemek için kullanılan eşik değeri
100) #bulunan köşeler arasında olabilecek min piksel sayısı
corners = corners.astype(int) #köşe koordinatlarını integer değerlere çevirir

for corner in corners:
    x, y = corner.ravel() #koordinatları tek boyutlu hale getirdi ->[[x,y]]'den [x,y]'e
    cv2.circle(img, (x,y), 5, (255, 0, 0), -1)

for i in range(len(corners)): #tüm köşeleri sırayla alır
    for j in range(i + 1, len(corners)): #her köşe bir sonraki köşeyle bağlanır
        corner1 = tuple(corners[i].ravel()) #tuple ile (x,y) formatına çevrilir
        corner2 = tuple(corners[j].ravel()) 
        color = tuple(int(c) for c in np.random.randint(0, 256, size= 3)) #0-255 arasında rastgele BGR formatında renk üretir
        cv2.line(img, corner1, corner2, color, 1)

cv2.imshow('Chess Board',img)
cv2.waitKey(0)
cv2.destroyAllWindows()
```

## Tutorial 8
Görüntüde seçtiğim bir nesnenin algılanmasını sağladım. Nesneyi bounding box ile gösterdim. 

```python

import numpy as np
import cv2

img = cv2.imread('C:/Users/ukugu/Desktop/yoda_opencv/cocuk.png', 0)
template = cv2.imread('C:/Users/ukugu/Desktop/yoda_opencv/top.png', 0)
img2 = img.copy() #orijinal resmin kopyası, tüm işlemler kopya üzerinde yapılacak
h, w = template.shape #resmin genişlik ve yüksekliğini belirler

methods = [cv2.TM_CCOEFF, cv2.TM_CCOEFF_NORMED, cv2.TM_CCORR,
           cv2.TM_CCORR_NORMED, cv2.TM_SQDIFF, cv2.TM_SQDIFF_NORMED] #resim üzerinde aradığımız nesneyi bulmak için kullandığımız yöntemler

for method in methods:
    img = img.copy() #teker teker tüm methodları kopya üzerinde deneyecek

   result = cv2.matchTemplate(img2, template, method) #template'i img2 içinde arıyor
   min_val, max_val, min_loc, max_loc = cv2.minMaxLoc(result)
    if method in [cv2.TM_SQDIFF, cv2.TM_SQDIFF_NORMED]:
        location = min_loc
    else:
        location = max_loc

   bottom_right = location[0] + w, location[1] + h
    cv2.rectangle(img2, location, bottom_right, 255, 5)
    cv2.imshow('Maç', img2)
    cv2.waitKey(0)
    cv2.destroyAllWindows()
```
